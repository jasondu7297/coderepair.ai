bool cmp(string s, string ss) {

  for (i = 1; i < s.size(); i += 2) s2 += s[i];
  for (i = 0; i < ss.size(); i += 2) ss1 += ss[i];
  for (i = 1; i < ss.size(); i += 2) ss2 += ss[i];

  return s2 > ss2;
}
