t = int(input())
for i in range(t):
    n = int(input())
    s = input().split()
    for j in range(n):
        s[j] = int(s[j])
    s.sort()
    hieu = 0
    res = s[0]
    for i in range(1,n):
        if (s[i-1]<0) or (s[i]-hieu-s[i-1]>=s[i-1]):
            s[i] = s[i] - hieu -s[i-1]
            hieu+=s[i-1]
            res =  s[i]
        else:
            break
    print(res)
