for _ in range(int(input())):
    n=int(input())
    ans=-1e10
    if(n==1):
        ans=input()
    else:
        arr=list(map(int, input().split()))
        #print(arr)
        arr.sort()
        #print(arr)
        
        for i in range(1,n):
            h=arr[i]-arr[i-1]
            if ans<=h:
                #print("debug",h, ans)
                ans=h
    print(ans)
