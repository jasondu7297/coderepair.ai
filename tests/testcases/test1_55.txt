t=int(input())
for i in range(t) :
    n=int(input())
    araye=input().split()
    for ii in range(n) :
        araye[ii]=int(araye[ii])
    araye.sort()
    mini=araye[0]
    while (len(araye) > 1) :
        if (mini >(araye[1]-araye[0])) and araye[0] >= 0:
            break
        adad=araye[0]
        for j in range (len(araye)) :
            araye[j]= araye[j] - adad
        araye.pop(0)
        araye.sort()
        if mini< araye[0] :
            mini=araye[0]    
    if mini< araye[0] :
            mini=araye[0]
    print(mini)
