

def solve(a, n):
    a.sort()
    a = a[::-1]
    c = a[-1]
    #print(a)
    while True:
        if len(a) == 1 or (a[0] - a[-1]) < a[-1]:
            return c
        else:
            r = a.pop()
            c = a[-1] - r


for _ in range(int(input())):
    n = int(input())
    a = list(map(int, input().split()))
    print(solve(a, n))
