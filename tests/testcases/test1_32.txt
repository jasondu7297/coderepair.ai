void solve()
{

   {
      ll x;
      cin>>x;
      s.insert(x);
   }
   if(s.size()==1)
   {
      cout<<*s.begin()<<endl;
      return ;
   }
   ll ans=*s.begin();
   s.erase(ans);
   ll del=0;
   ll curr=ans;
  
   while(!s.empty())
   {
         ans=max(ans,*s.begin()-curr);
            curr+=(*s.begin()-curr);
         //del+=(*s.begin()-del-curr);
         //curr=*s.begin();
         s.erase(*s.begin());
   }
   cout<<ans<<endl;
}
