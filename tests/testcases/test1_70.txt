for _ in range(int(input())):
    l = int(input())
    a = input().split()
    t = 0
    for i in range(len(a)):
        a[i] = int(a[i])
    if len(a) == 1:
        print(a[0])
        continue

    a.sort()

    if a[0] <= 0:
        t -= a[0]
        del a[0]

    if len(a) == 1:
        print(a[0] + t)
        continue
    m = a[0]
    for i in range(len(a) - 1):
        m = max(m,a[i+1] - a[i])

    print(m)
