for _ in range(it()):
	a = it()
	b = list(mp())
	b.sort()
	if len(b)==1:
		print(b[0])
		continue
	prev = 0
	ans =None
	for i in range(a):
		prev+=b[i]
		if i!=a-1:
			if (b[i+1]-prev)<b[i]:
				ans = b[i]
				break
			else:
				b[i+1]-=prev

	if ans ==None :
		print(b[-1])
	else:
		print(ans)
