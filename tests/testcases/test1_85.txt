for test in range(t):
    n = int(input())
    p = list(map(int,input().split()))
    if(len(p)==1):
        print(p[0])
        continue
    elif(len(p)==2):
        print(abs(p[0]-p[1]))
        continue
    ans = 0
    p.sort()
    for i in range(len(p)-2,0,-1):
        ans = max(ans,p[i+1]-p[i])
    print(ans)
