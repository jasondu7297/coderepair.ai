from heapq import heapify, heappush, heappop

t = int(input())

while t>0:
    n = int(input())
    a = list(map(int, input().split()))
    heapify(a)
    min = a[0]
    for i in range(n-1):
        e = heappop(a)
        if e>min:
            min = e
        for j in range(n-i-1):
            a[j]-=e
        heapify(a)
    print(min)
    t-=1
