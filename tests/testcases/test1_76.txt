def solve():
    n = int(input())
    l = input()
    l = [int(i) for i in l.split()]
    try:
      if (n == 1):
        print(l[0])
        return
      elif(n == 2):
        a = min(l)
        l.remove(a)
        a = max(a, l[0] - a)
        print(a)
        return
      else:
        l.remove(min(l))
        a = min(l)
        l.remove(a)
        b = min(l)
        print(b - a)
        return

    except EOFError as e:
      print("Error")


for case in range(int(input())):
  solve()
