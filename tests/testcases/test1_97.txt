import sys
readline = sys.stdin.readline

T = int(readline())

for t in range(T):
  N = int(readline())
  A = list(map(int,readline().split()))
  A = sorted(A)
  
  minus_sum = 0
  best = A[0]
  
  for i in range(N - 1):
    minus_sum += A[i]
    A[i + 1] -= minus_sum
    if A[i + 1] < best:
      break
    else:
      best = A[i + 1]
  print(best)
